 **1.3.4**
 ```java
 public class Test
{
	public static void main(String[] args)
	{
		String string="[()]{}{[()()]()}";
		if(isLegal(string))
			System.out.println("true");
		else
			System.out.println("flase");
	}
	public static boolean isLegal(String string)
	{
		Stack<Character> stack=new Stack<>();
		for (int i = 0; i <string.length() ; i++) 
		{
			char c=string.charAt(i);
			if(c=='[' ||c=='('||c=='{')
				stack.push(c);
			else if(c==')')//注意优先级
			{
				char temp=stack.pop();
				if(temp!='(')
					return false;
			}
			else if(c==']')
			{
				char temp=stack.pop();
				if(temp!='[')
					return false;
			}
			else if(c=='}')
			{
				char temp=stack.pop();
				if(temp!='{')  
					return false;
			}
		}
		if(!stack.isEmpty())
			return false;
		return true;
	}
}
```
