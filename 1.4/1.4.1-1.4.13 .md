```java
public class BinarySearch
{
	public static void main(String[] args)
	{
		int[] a = new int[10];
		for (int i = 0; i < a.length; i++)
		{
			a[i] = i % 5;
		}
		Arrays.sort(a);
		for (int i = 0; i < a.length; i++) 
		{
			System.out.print(a[i]+" ");
		}
		System.out.println();
		System.out.println(rank(1, a));
	}

	public static int rank(int key, int[] a)
	{
		return rank(key, a, 0, a.length-1);
	}

	public static int rank(int key, int[] a, int lo, int hi)
	{
		if(hi<lo)
			return -1;
		int mid=lo+(hi-lo)/2;
		if(a[mid]<key)
			return rank(key, a, mid+1, hi);
		else if(a[mid]>key)
			return rank(key, a, lo, mid-1);
		else
		{
			while(mid-1>=0 && a[mid]==a[mid-1])
			{
				mid--;
			}
			return mid;
		}
	}
}
```

```java
public class StaticSETofInts
{
	private int[] a;

	public StaticSETofInts(int[] keys)
	{
		a = new int[keys.length];
		for (int i = 0; i < keys.length; i++)
			a[i] = keys[i];
		Arrays.sort(a);
	}

	public boolean contains(int key)
	{
		return rank(key) != -1;
	}

	public int howMany(int key)
	{
		int mid=rank(key);
		if(mid == -1)
			return 0;
		int count=1;
		//用了上一题的内容,所以count得到的是最左边的匹配的值
		while(mid+1<a.length && a[mid]==a[mid+1] )
		{
			mid++;
			count++;
		}
		return count;
	}

	public int rank(int key)
	{
		int lo = 0;
		int hi = a.length - 1;
		while (lo <= hi)
		{
			// Key is in a[lo..hi] or not present.
			int mid = lo + (hi - lo) / 2;
			if (key < a[mid])
				hi = mid - 1;
			else if (key > a[mid])
				lo = mid + 1;
			else
			{
				while (mid - 1 >= 0 && a[mid] == a[mid - 1])
				{
					mid--;
				}
				return mid;
			}
		}
		return -1;
	}
	public static void main(String[] args)
	{
		int [] keys= {2,1,3,3,3,5,4};
		StaticSETofInts s=new StaticSETofInts(keys);
		System.out.println(s.howMany(3));
	}
}
```
